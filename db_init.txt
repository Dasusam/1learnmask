drop table if exists air_quality10;
drop table if exists air_quality25;
drop table if exists covid;

CREATE TABLE air_quality10
                (id INTEGER PRIMARY KEY AUTOINCREMENT,
                 dataTime TEXT,
                 pm10Value INTEGER);

CREATE TABLE air_quality25
                (id INTEGER PRIMARY KEY AUTOINCREMENT,
                 dataTime TEXT,
                 pm25Value INTEGER);

CREATE TABLE covid
                (id INTEGER PRIMARY KEY AUTOINCREMENT,
                 date TEXT,
                 value INTEGER);
